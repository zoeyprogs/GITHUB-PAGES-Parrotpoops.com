                #include 
                using namespace std;
                
                // represents a point on a standard Cartesian plane

                class Point {
                private:
                    double x;
                    double y;
                
                public:
                    Point() {
                        x = 0;
                        y = 0;
                    }
                
                    double getX() {
                        return x;
                    }
                    
                    double getY() {
                        return y;
                    }
                    
                    void translate(double deltaX, double deltaY) {
                        x += deltaX;
                        y += deltaY;
                    }
                };
                
                int main()
                {
                    Point p1;
                    Point p2;
                    
                    // move Point p1: right 10 units, down 3 units
                    // move Point p2: left 2 units, up 5 units
                    
                    p1.translate(10,-3);
                    p2.translate(-2,5);
                    
                    cout << p1.getX() << " " << p1.getY() << endl;
                    cout << p2.getX() << " " << p2.getY() << endl;
                }

                {

                    Point p1(rand() % 10, rand() % 10);
                    
                    // create a point p2: right 4 units of p1, down 3 units from it
                    
                    double x1;
                    double y1;
                    
                    Point p2;

                    double xShift = x1 + 4;
                    double yShift = y1 - 3;
                    
                    p2.translate(xShift, yShift);

                #include 
                using namespace std;
                
                class Rectangle {
                public:
                    Rectangle(double theWidth, double theHeight);
                    double getWidth();
                    double getHeight();
                    double getPerimeter();
                    double getArea();
                    void scale(double scaleFactor);
                
                private:
                    double width;
                    double height;
                };
                
                
                // Add the function implementations so the code compiles and runs successfully
                

                Rectangle::Rectangle(double theWidth, double theHeight) {
                    width = theWidth;
                    height = theHeight;
                }
                
                double Rectangle::getWidth() {
                    return width;
                }
                
                double Rectangle::getHeight() {
                    return height;
                }
                
                double Rectangle::getPerimeter() {
                    return (width * 2) + (height * 2);
                }
                
                double Rectangle::getArea() {
                    return width * height;
                }
                
                void Rectangle::scale(double scaleFactor) {
                    width *= scaleFactor;
                    height = height * scaleFactor;
                }
                
                int main()
                {
                    // create a rectangle with a width of 10 and a height of 20
                    
                    Rectangle r1(10, 20);
                
                    cout << r1.getWidth() << " " << r1.getHeight() << endl;
                    cout << r1.getPerimeter() << endl;
                    cout << r1.getArea() << endl;
                
                
                    // reduce the rectangle to half its former size
                    
                    r1.scale(0.5);
                
                    cout << r1.getWidth() << " " << r1.getHeight() << endl;
                    cout << r1.getPerimeter() << endl;
                    cout << r1.getArea() << endl;
                }
